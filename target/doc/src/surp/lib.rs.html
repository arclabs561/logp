1//! # surp
2//!
3//! Information theory primitives: surprisal, entropy, KL divergence.
4//!
5//! (surp: from "surprisal", the fundamental unit of information) If an event
6//! is certain, learning it happened conveys no information. If it's rare,
7//! learning it happened is highly informative.
8//!
9//! ## Key Functions
10//!
11//! | Function | Measures | Formula |
12//! |----------|----------|---------|
13//! | [`entropy`] | Uncertainty in distribution | H(p) = -Σ p(x) log p(x) |
14//! | [`cross_entropy`] | Expected bits using wrong code | H(p,q) = -Σ p(x) log q(x) |
15//! | [`kl_divergence`] | "Distance" from q to p | KL(p‖q) = Σ p(x) log(p(x)/q(x)) |
16//! | [`js_divergence`] | Symmetric "distance" | JS(p,q) = ½KL(p‖m) + ½KL(q‖m) |
17//! | [`mutual_information`] | Shared information | I(X;Y) = H(X) + H(Y) - H(X,Y) |
18//!
19//! ## Quick Start
20//!
21//! ```rust
22//! use surp::{entropy, kl_divergence, js_divergence};
23//!
24//! let p = [0.25, 0.25, 0.25, 0.25];  // uniform
25//! let q = [0.5, 0.25, 0.125, 0.125]; // skewed
26//!
27//! let h = entropy(&p);           // 2.0 bits (maximum for 4 outcomes)
28//! let kl = kl_divergence(&p, &q); // asymmetric divergence
29//! let js = js_divergence(&p, &q); // symmetric, bounded [0, 1]
30//! ```
31//!
32//! ## Why These Matter for ML
33//!
34//! - **Cross-entropy loss**: Standard classification loss is cross-entropy
35//!   between true labels (one-hot) and predicted probabilities
36//! - **KL divergence**: VAE latent space regularization, knowledge distillation
37//! - **Mutual information**: Feature selection, representation learning bounds
38//! - **JS divergence**: GAN training (original formulation)
39//!
40//! ## Connections
41//!
42//! - [`kern`](../kern): MMD and KL both measure distribution "distance"
43//! - [`wass`](../wass): Wasserstein vs entropy-based divergences
44//! - [`stratify`](../stratify): NMI for cluster evaluation uses this crate
45//! - [`fynch`](../fynch): Temperature scaling affects entropy calibration
46//!
47//! ## References
48//!
49//! - Shannon (1948). "A Mathematical Theory of Communication"
50//! - Cover & Thomas (2006). "Elements of Information Theory"
51//!
52//! ## Sublinear Estimation
53//!
54//! The [`unseen`] module implements the Valiant-Valiant estimators
55//! for entropy and support size from sublinear samples.
56//!
57//! **Key insight**: Given O(k/log k) samples from a distribution over k elements,
58//! we can accurately estimate entropy, support size, and distances. This is
59//! optimal up to constant factors.
60//!
61//! **References**:
62//! - Valiant & Valiant (2011). "Estimating the Unseen: An n/log(n)-Sample Estimator"
63//! - Valiant & Valiant (2017). "Estimating the Unseen: Improved Estimators" (JACM)
64//!
65//! ## What Can Go Wrong
66//!
67//! 1. **Zero probabilities in KL**: KL(p||q) is infinite if q(x)=0 where p(x)>0.
68//!    Use JS divergence or add smoothing.
69//! 2. **Not normalized**: Many functions assume probabilities sum to 1. Check inputs.
70//! 3. **Entropy of continuous distributions**: Discrete entropy ≠ differential entropy.
71//!    Discretization bin width affects results.
72//! 4. **Sample size too small**: Entropy estimators are biased for small samples.
73//!    Miller-Madow correction or Valiant-Valiant for sublinear regime.
74//! 5. **Floating point issues**: Very small probabilities can cause log(0) = -inf.

115/// Compute the fingerprint of a sample: F[i] = count of elements appearing exactly i times.
116///
117/// The fingerprint captures the "shape" of a distribution without the labels.
118/// It's the key data structure for sublinear entropy estimation.
119///
120/// # Arguments
121///
122/// * `samples` - Slice of sample values (any Eq + Hash + Clone type)
123///
124/// # Returns
125///
126/// Vector where index i contains count of elements appearing exactly i+1 times.
127/// (Index 0 = elements appearing once, index 1 = appearing twice, etc.)
128///
129/// # Example
130///
131/// ```rust
132/// use surp::fingerprint;
133///
134/// let samples = ["a", "a", "b", "c", "c", "c"];
135/// let fp = fingerprint(&samples);
136///
137/// // "b" appears 1 time, "a" appears 2 times, "c" appears 3 times
138/// assert_eq!(fp[0], 1);  // F_1: one element appears once
139/// assert_eq!(fp[1], 1);  // F_2: one element appears twice  
140/// assert_eq!(fp[2], 1);  // F_3: one element appears three times
141/// ```
142pub fn fingerprint<T: std::hash::Hash + Eq + Clone>(samples: &[T]) -> Vec<usize> {
143    use std::collections::HashMap;
144
145    // Count occurrences
146    let mut counts: HashMap<T, usize> = HashMap::new();
147    for s in samples {
148        *counts.entry(s.clone()).or_insert(0) += 1;
149    }
150
151    // Build fingerprint
152    let max_count = counts.values().copied().max().unwrap_or(0);
153    let mut fp = vec![0usize; max_count];
154
155    for &count in counts.values() {
156        if count > 0 {
157            fp[count - 1] += 1;
158        }
159    }
160
161    fp
162}

164/// Count distinct elements in a sample.
165///
166/// This is the naive estimator - actual support size may be larger.
167pub fn distinct_count<T: std::hash::Hash + Eq>(samples: &[T]) -> usize {
168    use std::collections::HashSet;
169    samples.iter().collect::<HashSet<_>>().len()
170}

172/// Good-Turing estimate of unseen probability mass.
173///
174/// Estimates the total probability mass of elements not seen in the sample.
175/// Uses the simple formula: P(unseen) ≈ F₁ / n where F₁ is the count of
176/// elements appearing exactly once (singletons) and n is sample size.
177///
178/// # Arguments
179///
180/// * `fingerprint` - The fingerprint of the sample
181/// * `sample_size` - Total number of samples
182///
183/// # Returns
184///
185/// Estimated probability mass of unseen elements
186///
187/// # Example
188///
189/// ```rust
190/// use surp::{fingerprint, good_turing_unseen_mass};
191///
192/// let samples: Vec<u32> = vec![1, 1, 2, 3, 4, 5, 5, 5];
193/// let fp = fingerprint(&samples);
194/// let unseen = good_turing_unseen_mass(&fp, samples.len());
195///
196/// // F_1 = 2 (elements 2,3,4 appear once - wait, that's 3)
197/// // Actually: 2 appears 1x, 3 appears 1x, 4 appears 1x = F_1 = 3
198/// // unseen ≈ 3/8 = 0.375
199/// assert!(unseen > 0.0);
200/// ```
201pub fn good_turing_unseen_mass(fingerprint: &[usize], sample_size: usize) -> f64 {
202    if sample_size == 0 || fingerprint.is_empty() {
203        return 0.0;
204    }
205    fingerprint[0] as f64 / sample_size as f64
206}

208/// Shannon entropy: H(p) = -Σ p(x) log₂ p(x)
209///
210/// Measures the average "surprise" or uncertainty in a distribution.
211/// Maximum entropy = log₂(n) for uniform distribution over n outcomes.
212///
213/// # Arguments
214///
215/// * `p` - Probability distribution (must sum to ~1.0)
216///
217/// # Returns
218///
219/// Entropy in bits (base-2 logarithm)
220///
221/// # Example
222///
223/// ```rust
224/// use surp::entropy;
225///
226/// let uniform = [0.25, 0.25, 0.25, 0.25];
227/// let certain = [1.0, 0.0, 0.0, 0.0];
228///
229/// assert!((entropy(&uniform) - 2.0).abs() < 1e-10);  // max entropy
230/// assert!(entropy(&certain) < 1e-10);                 // zero entropy
231/// ```
232pub fn entropy(p: &[f64]) -> f64 {
233    p.iter()
234        .filter(|&&x| x > EPSILON)
235        .map(|&x| -x * x.ln() / LN_2)
236        .sum()
237}

239/// Cross-entropy: H(p, q) = -Σ p(x) log₂ q(x)
240///
241/// Expected number of bits needed to encode samples from p using
242/// a code optimized for q. Always ≥ H(p), with equality iff p = q.
243///
244/// # Arguments
245///
246/// * `p` - True distribution
247/// * `q` - Model distribution
248///
249/// # Returns
250///
251/// Cross-entropy in bits
252///
253/// # Example
254///
255/// ```rust
256/// use surp::{entropy, cross_entropy};
257///
258/// let p = [0.5, 0.5];
259/// let q = [0.9, 0.1];
260///
261/// let h_p = entropy(&p);
262/// let h_pq = cross_entropy(&p, &q);
263/// assert!(h_pq >= h_p);  // cross-entropy ≥ entropy
264/// ```
265pub fn cross_entropy(p: &[f64], q: &[f64]) -> f64 {
266    assert_eq!(p.len(), q.len(), "distributions must have same length");
267
268    p.iter()
269        .zip(q.iter())
270        .filter(|(&pi, _)| pi > EPSILON)
271        .map(|(&pi, &qi)| -pi * (qi.max(EPSILON)).ln() / LN_2)
272        .sum()
273}

275/// KL divergence: KL(p ‖ q) = Σ p(x) log₂(p(x) / q(x))
276///
277/// Measures how much information is lost when q is used to approximate p.
278/// **Not symmetric**: KL(p‖q) ≠ KL(q‖p).
279///
280/// # Properties
281///
282/// - KL(p‖q) ≥ 0 (Gibbs' inequality)
283/// - KL(p‖q) = 0 iff p = q
284/// - Undefined if q(x) = 0 where p(x) > 0
285///
286/// # Arguments
287///
288/// * `p` - True distribution
289/// * `q` - Approximate distribution
290///
291/// # Returns
292///
293/// KL divergence in bits
294///
295/// # Example
296///
297/// ```rust
298/// use surp::kl_divergence;
299///
300/// let p = [0.5, 0.5];
301/// let q = [0.9, 0.1];
302///
303/// let kl_pq = kl_divergence(&p, &q);
304/// let kl_qp = kl_divergence(&q, &p);
305///
306/// assert!(kl_pq >= 0.0);
307/// assert!((kl_pq - kl_qp).abs() > 0.1);  // asymmetric!
308/// ```
309pub fn kl_divergence(p: &[f64], q: &[f64]) -> f64 {
310    assert_eq!(p.len(), q.len(), "distributions must have same length");
311
312    p.iter()
313        .zip(q.iter())
314        .filter(|(&pi, _)| pi > EPSILON)
315        .map(|(&pi, &qi)| pi * (pi / qi.max(EPSILON)).ln() / LN_2)
316        .sum()
317}

319/// Jensen-Shannon divergence: JS(p, q) = ½KL(p‖m) + ½KL(q‖m)
320///
321/// Symmetric, bounded version of KL divergence where m = (p + q) / 2.
322///
323/// # Properties
324///
325/// - 0 ≤ JS(p,q) ≤ 1 (in bits)
326/// - JS(p,q) = JS(q,p) (symmetric)
327/// - √JS is a proper metric
328///
329/// # Arguments
330///
331/// * `p` - First distribution
332/// * `q` - Second distribution
333///
334/// # Returns
335///
336/// JS divergence in bits, bounded [0, 1]
337///
338/// # Example
339///
340/// ```rust
341/// use surp::js_divergence;
342///
343/// let p = [0.5, 0.5];
344/// let q = [0.9, 0.1];
345///
346/// let js = js_divergence(&p, &q);
347/// assert!(js >= 0.0 && js <= 1.0);
348/// assert!((js_divergence(&p, &q) - js_divergence(&q, &p)).abs() < 1e-10);
349/// ```
350pub fn js_divergence(p: &[f64], q: &[f64]) -> f64 {
351    assert_eq!(p.len(), q.len(), "distributions must have same length");
352
353    let m: Vec<f64> = p
354        .iter()
355        .zip(q.iter())
356        .map(|(&pi, &qi)| (pi + qi) / 2.0)
357        .collect();
358
359    0.5 * kl_divergence(p, &m) + 0.5 * kl_divergence(q, &m)
360}

362/// Mutual information from joint distribution: I(X;Y) = H(X) + H(Y) - H(X,Y)
363///
364/// Measures how much knowing X reduces uncertainty about Y (and vice versa).
365///
366/// # Arguments
367///
368/// * `joint` - Joint probability matrix P(X,Y), flattened row-major
369/// * `rows` - Number of X outcomes
370/// * `cols` - Number of Y outcomes
371///
372/// # Returns
373///
374/// Mutual information in bits
375///
376/// # Example
377///
378/// ```rust
379/// use surp::mutual_information;
380///
381/// // Independent: P(X,Y) = P(X)P(Y)
382/// let independent = [0.25, 0.25, 0.25, 0.25];  // uniform 2x2
383///
384/// // Perfectly dependent: X = Y
385/// let dependent = [0.5, 0.0, 0.0, 0.5];
386///
387/// let mi_indep = mutual_information(&independent, 2, 2);
388/// let mi_dep = mutual_information(&dependent, 2, 2);
389///
390/// assert!(mi_indep < 0.01);  // ~0 for independent
391/// assert!(mi_dep > 0.9);     // ~1 bit for perfect dependence
392/// ```
393pub fn mutual_information(joint: &[f64], rows: usize, cols: usize) -> f64 {
394    assert_eq!(
395        joint.len(),
396        rows * cols,
397        "joint size must match rows * cols"
398    );
399
400    // Marginals
401    let mut p_x = vec![0.0; rows];
402    let mut p_y = vec![0.0; cols];
403
404    for i in 0..rows {
405        for j in 0..cols {
406            let p_xy = joint[i * cols + j];
407            p_x[i] += p_xy;
408            p_y[j] += p_xy;
409        }
410    }
411
412    // I(X;Y) = Σ P(x,y) log(P(x,y) / (P(x)P(y)))
413    let mut mi = 0.0;
414    for i in 0..rows {
415        for j in 0..cols {
416            let p_xy = joint[i * cols + j];
417            if p_xy > EPSILON && p_x[i] > EPSILON && p_y[j] > EPSILON {
418                mi += p_xy * (p_xy / (p_x[i] * p_y[j])).ln() / LN_2;
419            }
420        }
421    }
422
423    mi.max(0.0) // Numerical errors can make it slightly negative
424}

426/// Conditional entropy: H(Y|X) = H(X,Y) - H(X)
427///
428/// Average uncertainty remaining in Y after observing X.
429///
430/// # Arguments
431///
432/// * `joint` - Joint probability matrix P(X,Y), flattened row-major
433/// * `rows` - Number of X outcomes
434/// * `cols` - Number of Y outcomes
435///
436/// # Returns
437///
438/// Conditional entropy in bits
439pub fn conditional_entropy(joint: &[f64], rows: usize, cols: usize) -> f64 {
440    let h_xy = entropy(joint);
441
442    // Marginal H(X)
443    let mut p_x = vec![0.0; rows];
444    for i in 0..rows {
445        for j in 0..cols {
446            p_x[i] += joint[i * cols + j];
447        }
448    }
449    let h_x = entropy(&p_x);
450
451    h_xy - h_x
452}

454/// Normalized mutual information: NMI(X,Y) = 2*I(X;Y) / (H(X) + H(Y))
455///
456/// Scaled to [0, 1] for easier interpretation and comparison.
457///
458/// # Arguments
459///
460/// * `joint` - Joint probability matrix P(X,Y), flattened row-major
461/// * `rows` - Number of X outcomes
462/// * `cols` - Number of Y outcomes
463///
464/// # Returns
465///
466/// NMI in [0, 1]
467pub fn normalized_mutual_information(joint: &[f64], rows: usize, cols: usize) -> f64 {
468    // Marginals
469    let mut p_x = vec![0.0; rows];
470    let mut p_y = vec![0.0; cols];
471
472    for i in 0..rows {
473        for j in 0..cols {
474            let p_xy = joint[i * cols + j];
475            p_x[i] += p_xy;
476            p_y[j] += p_xy;
477        }
478    }
479
480    let h_x = entropy(&p_x);
481    let h_y = entropy(&p_y);
482
483    if h_x + h_y < EPSILON {
484        return 1.0; // Both constant → perfectly "correlated"
485    }
486
487    let mi = mutual_information(joint, rows, cols);
488    2.0 * mi / (h_x + h_y)
489}
