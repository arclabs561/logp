1//! # Sublinear Estimation via the Unseen
2//!
3//! Implements the Valiant-Valiant estimators for distribution properties
4//! from sublinear samples.
5//!
6//! ## The Key Insight
7//!
8//! Given n samples from a distribution over k elements, the naive approach
9//! uses O(k) samples. Valiant & Valiant showed that O(k/log k) samples suffice
10//! for accurate estimation of:
11//!
12//! - **Entropy**: H(p) = -Σ p(x) log p(x)
13//! - **Support size**: |{x : p(x) > 0}|
14//! - **Distance**: Total variation between two distributions
15//!
16//! ## The Algorithm
17//!
18//! 1. Compute the **fingerprint** F where F[i] = count of elements seen i times
19//! 2. Find the "simplest" histogram h that could have generated F via linear programming
20//! 3. Estimate properties from h
21//!
22//! The LP objective minimizes support size (Occam's razor), subject to
23//! expected fingerprint ≈ observed fingerprint.
24//!
25//! ## Complexity
26//!
27//! - Sample complexity: O(k / log k) for distributions over k elements
28//! - Time complexity: O(n log n) for LP setup + O(poly(log k)) for LP solve
29//!
30//! ## References
31//!
32//! - Valiant & Valiant (2011). "Estimating the Unseen: An n/log(n)-Sample Estimator"
33//! - Valiant & Valiant (2017). "Estimating the Unseen: Improved Estimators" (JACM)
34

38/// Configuration for the unseen estimators.
39#[derive(Clone, Debug)]
40pub struct UnseenConfig {
41    /// Geometric ratio for probability grid (default: 1.5)
42    pub grid_ratio: f64,
43    /// Number of fingerprint entries to fit (default: 10)
44    pub fit_count: usize,
45    /// Slack factor for constraints (default: 1.0)
46    pub slack_factor: f64,
47}
48
49impl Default for UnseenConfig {
50    fn default() -> Self {
51        Self {
52            grid_ratio: 1.5,
53            fit_count: 10,
54            slack_factor: 1.0,
55        }
56    }
57}

59/// Estimate entropy of a distribution including unseen elements.
60///
61/// Uses the linear programming approach from Valiant & Valiant to reconstruct
62/// a plausible histogram consistent with the observed fingerprint.
63///
64/// # Arguments
65///
66/// * `samples` - Sample data from the distribution
67///
68/// # Returns
69///
70/// Estimated entropy in bits
71///
72/// # Example
73///
74/// ```rust
75/// use surp::unseen::entropy_unseen;
76///
77/// // Sample 50 distinct elements once each (uniform-ish)
78/// let samples: Vec<u32> = (0..50).collect();
79/// let entropy = entropy_unseen(&samples);
80///
81/// // For uniform over ~100 elements, entropy ≈ log2(100) ≈ 6.6 bits
82/// // Our estimate should be > log2(50) ≈ 5.6 bits
83/// assert!(entropy > 5.0);
84/// ```
85pub fn entropy_unseen<T: std::hash::Hash + Eq + Clone>(samples: &[T]) -> f64 {
86    entropy_unseen_with_config(samples, &UnseenConfig::default())
87}

89/// Entropy estimation with custom configuration.
90pub fn entropy_unseen_with_config<T: std::hash::Hash + Eq + Clone>(
91    samples: &[T],
92    config: &UnseenConfig,
93) -> f64 {
94    let fp = fingerprint(samples);
95    let n = samples.len() as f64;
96
97    let hist = recover_histogram_with_config(&fp, n, config);
98
99    // H = -Σ count * prob * log(prob)
100    hist.iter()
101        .map(|&(prob, count)| {
102            if prob > 1e-12 {
103                -prob * prob.ln() / std::f64::consts::LN_2 * count
104            } else {
105                0.0
106            }
107        })
108        .sum()
109}

111/// Estimate support size including unseen elements.
112///
113/// This estimates the true number of distinct elements in the underlying
114/// distribution, even those not seen in the sample.
115///
116/// # Arguments
117///
118/// * `samples` - Sample data from the distribution
119///
120/// # Returns
121///
122/// Estimated support size (number of distinct elements)
123///
124/// # Example
125///
126/// ```rust
127/// use surp::unseen::support_unseen;
128///
129/// // If we see 50 elements each exactly once, there are likely more
130/// let samples: Vec<u32> = (0..50).collect();
131/// let support = support_unseen(&samples);
132///
133/// assert!(support >= 50.0);  // At least what we saw
134/// ```
135pub fn support_unseen<T: std::hash::Hash + Eq + Clone>(samples: &[T]) -> f64 {
136    support_unseen_with_config(samples, &UnseenConfig::default())
137}

139/// Support estimation with custom configuration.
140pub fn support_unseen_with_config<T: std::hash::Hash + Eq + Clone>(
141    samples: &[T],
142    config: &UnseenConfig,
143) -> f64 {
144    let fp = fingerprint(samples);
145    let n = samples.len() as f64;
146
147    let hist = recover_histogram_with_config(&fp, n, config);
148    hist.iter().map(|&(_, count)| count).sum()
149}

151/// Estimate total variation distance between two distributions from samples.
152///
153/// Uses a heuristic based on support overlap. For full Valiant-Valiant
154/// distance estimation, a 2D fingerprint LP would be needed.
155///
156/// # Arguments
157///
158/// * `samples_p` - Samples from distribution P
159/// * `samples_q` - Samples from distribution Q
160///
161/// # Returns
162///
163/// Estimated distance heuristic (higher = more different)
164///
165/// # Example
166///
167/// ```rust
168/// use surp::unseen::distance_unseen;
169///
170/// // Samples from different ranges (no overlap)
171/// let p: Vec<u32> = (0..50).collect();
172/// let r: Vec<u32> = (100..150).collect();
173///
174/// let dist = distance_unseen(&p, &r);
175/// // With disjoint samples, distance should be positive
176/// assert!(dist >= 0.0);
177/// ```
178pub fn distance_unseen<T: std::hash::Hash + Eq + Clone, U: std::hash::Hash + Eq + Clone>(
179    samples_p: &[T],
180    samples_q: &[U],
181) -> f64 {
182    // For now, use a simpler approach based on individual histograms
183    // Full Valiant-Valiant uses 2D fingerprint which is more complex
184
185    let fp_p = fingerprint(samples_p);
186    let fp_q = fingerprint(samples_q);
187    let n_p = samples_p.len() as f64;
188    let n_q = samples_q.len() as f64;
189
190    let config = UnseenConfig::default();
191    let hist_p = recover_histogram_with_config(&fp_p, n_p, &config);
192    let hist_q = recover_histogram_with_config(&fp_q, n_q, &config);
193
194    // Estimate total probability mass at each probability level
195    let mass_p: f64 = hist_p.iter().map(|&(prob, count)| prob * count).sum();
196    let mass_q: f64 = hist_q.iter().map(|&(prob, count)| prob * count).sum();
197
198    // Support sizes
199    let support_p: f64 = hist_p.iter().map(|&(_, count)| count).sum();
200    let support_q: f64 = hist_q.iter().map(|&(_, count)| count).sum();
201
202    // Rough bound: if supports are very different, distance is large
203    let support_overlap = support_p.min(support_q) / support_p.max(support_q);
204
205    // Heuristic combination (proper implementation would use 2D LP)
206    ((1.0 - support_overlap) + (mass_p - mass_q).abs()) / 2.0
207}

209/// Recover the underlying histogram from a sample fingerprint.
210///
211/// Returns a vector of (probability, count) pairs.
212/// For example, `[(0.1, 5.0)]` means "5 elements have probability 0.1".
213///
214/// This is the core of the Valiant-Valiant approach:
215///
216/// 1. Define a grid of possible probability values
217/// 2. Solve an LP to find the simplest histogram that explains the fingerprint
218/// 3. "Simplest" = minimum support size (Occam's razor)
219pub fn recover_histogram(fingerprint: &[usize], n: f64) -> Vec<(f64, f64)> {
220    recover_histogram_with_config(fingerprint, n, &UnseenConfig::default())
221}

223/// Histogram recovery with custom configuration.
224pub fn recover_histogram_with_config(
225    fingerprint: &[usize],
226    n: f64,
227    config: &UnseenConfig,
228) -> Vec<(f64, f64)> {
229    if fingerprint.is_empty() || n < 1.0 {
230        return Vec::new();
231    }
232
233    // Build probability grid
234    // Valiant suggests geometric spacing for the "unseen" region
235    let mut grid = Vec::new();
236
237    // Unseen region: very small probabilities
238    let mut x = 1.0 / (n * n);
239    while x <= 1.0 {
240        grid.push(x);
241        x *= config.grid_ratio;
242    }
243
244    // Seen region: explicit empirical probabilities
245    for i in 1..=fingerprint.len() {
246        grid.push(i as f64 / n);
247    }
248
249    grid.sort_by(|a, b| a.partial_cmp(b).unwrap());
250    grid.dedup_by(|a, b| (*a - *b).abs() < 1e-15);
251
252    // Setup LP: minimize support size subject to fingerprint constraints
253    let mut problem = Problem::new(OptimizationDirection::Minimize);
254
255    // Variables: h[j] = count of elements with probability grid[j]
256    let vars: Vec<_> = grid
257        .iter()
258        .map(|_| problem.add_var(1.0, (0.0, f64::INFINITY)))
259        .collect();
260
261    // Constraint: E[F_i] ≈ observed F_i for i = 1..fit_count
262    // E[F_i] = Σ_j h[j] * Poisson(n * grid[j], i)
263    let fit_k = fingerprint.len().min(config.fit_count);
264
265    for (i, &obs) in fingerprint.iter().enumerate().take(fit_k) {
266        let obs_count = obs as f64;
267        let k = (i + 1) as i32;
268
269        let mut expr = minilp::LinearExpr::empty();
270        for (j, &p) in grid.iter().enumerate() {
271            let poisson_prob = poisson_pmf(n * p, k);
272            expr.add(vars[j], poisson_prob);
273        }
274
275        // Slack proportional to sqrt(expected count) + constant
276        let slack = config.slack_factor * (obs_count.sqrt() + 1.0).max(1.0);
277
278        problem.add_constraint(expr.clone(), ComparisonOp::Ge, obs_count - slack);
279        problem.add_constraint(expr, ComparisonOp::Le, obs_count + slack);
280    }
281
282    // Constraint: total probability mass ≤ 1
283    let mut prob_expr = minilp::LinearExpr::empty();
284    for (j, &p) in grid.iter().enumerate() {
285        prob_expr.add(vars[j], p);
286    }
287    problem.add_constraint(prob_expr, ComparisonOp::Le, 1.0);
288
289    // Solve and extract solution
290    match problem.solve() {
291        Ok(solution) => grid
292            .iter()
293            .enumerate()
294            .filter_map(|(j, &p)| {
295                let count = solution[vars[j]];
296                if count > 1e-6 {
297                    Some((p, count))
298                } else {
299                    None
300                }
301            })
302            .collect(),
303        Err(_) => {
304            // Fallback: empirical histogram
305            fingerprint
306                .iter()
307                .enumerate()
308                .filter(|(_, &count)| count > 0)
309                .map(|(i, &count)| ((i + 1) as f64 / n, count as f64))
310                .collect()
311        }
312    }
313}

315/// Poisson probability mass function: P(X = k) = λ^k e^(-λ) / k!
316fn poisson_pmf(lambda: f64, k: i32) -> f64 {
317    if lambda <= 0.0 || k < 0 {
318        return if k == 0 && lambda == 0.0 { 1.0 } else { 0.0 };
319    }
320
321    let k_f = k as f64;
322    // Use log-space to avoid overflow: log(PMF) = k*log(λ) - λ - log(k!)
323    let log_pmf = k_f * lambda.ln() - lambda - ln_factorial(k as usize);
324    log_pmf.exp()
325}
326
327/// Compute ln(n!) using log-sum for small n, Stirling for large n
328fn ln_factorial(n: usize) -> f64 {
329    if n <= 20 {
330        // Direct computation for small values
331        (1..=n).map(|i| (i as f64).ln()).sum()
332    } else {
333        // Stirling's approximation: ln(n!) ≈ n*ln(n) - n + 0.5*ln(2πn)
334        let nf = n as f64;
335        nf * nf.ln() - nf + 0.5 * (2.0 * std::f64::consts::PI * nf).ln()
336    }
337}

339/// Estimate properties of a distribution with confidence bounds.
340///
341/// Returns (estimate, lower_bound, upper_bound) via bootstrap.
342pub fn entropy_unseen_with_ci<T: std::hash::Hash + Eq + Clone>(
343    samples: &[T],
344    confidence: f64,
345    num_bootstrap: usize,
346) -> (f64, f64, f64) {
347    use rand::seq::SliceRandom;
348    let mut rng = rand::thread_rng();
349
350    let point_estimate = entropy_unseen(samples);
351
352    if samples.len() < 10 || num_bootstrap == 0 {
353        return (point_estimate, point_estimate, point_estimate);
354    }
355
356    let mut bootstrap_estimates = Vec::with_capacity(num_bootstrap);
357
358    for _ in 0..num_bootstrap {
359        let resampled: Vec<_> = (0..samples.len())
360            .map(|_| samples.choose(&mut rng).unwrap().clone())
361            .collect();
362        bootstrap_estimates.push(entropy_unseen(&resampled));
363    }
364
365    bootstrap_estimates.sort_by(|a, b| a.partial_cmp(b).unwrap());
366
367    let alpha = (1.0 - confidence) / 2.0;
368    let lower_idx = (alpha * num_bootstrap as f64) as usize;
369    let upper_idx = ((1.0 - alpha) * num_bootstrap as f64) as usize;
370
371    (
372        point_estimate,
373        bootstrap_estimates[lower_idx.min(bootstrap_estimates.len() - 1)],
374        bootstrap_estimates[upper_idx.min(bootstrap_estimates.len() - 1)],
375    )
376}
