1//! # f-Divergences
2//!
3//! Generalized divergences between probability distributions.
4//!
5//! ## The f-Divergence Framework
6//!
7//! All f-divergences have the form:
8//!
9//! D_f(P || Q) = Σ q(x) f(p(x)/q(x))
10//!
11//! where f is a convex function with f(1) = 0.
12//!
13//! ## Key Divergences
14//!
15//! | Divergence | f(t) | Properties |
16//! |------------|------|------------|
17//! | KL | t log t | Asymmetric, unbounded |
18//! | Reverse KL | -log t | Mode-seeking |
19//! | χ² | (t-1)² | Squared difference |
20//! | Hellinger | (√t - 1)² | Symmetric, bounded [0,2] |
21//! | Total Variation | |t-1|/2 | Metric, bounded [0,1] |
22//! | α-divergence | (t^α - 1)/(α(α-1)) | Interpolates KL |
23//!
24//! ## Connections
25//!
26//! - f-divergences are convex in (P, Q) → useful for optimization
27//! - Variational bounds: D_f(P||Q) ≥ sup_T E_P[T] - E_Q[f*(T)]
28//! - This is the basis for f-GAN training objectives
29//!
30//! ## Statistical Estimation
31//!
32//! Hellinger distance and Fisher information play key roles in statistical minimax theory.
33//! The two-point testing method uses H²(P,Q) to derive lower bounds: if H² = O(1/n),
34//! then n samples cannot distinguish P from Q reliably.
35//!
36//! - Hellinger: tensorizes nicely for i.i.d. samples → [`hellinger_squared_tensorized`]
37//! - Fisher: governs Cramér-Rao bound → [`fisher_distance_approx`]
38
39use std::f64::consts::LN_2;

41const EPSILON: f64 = 1e-12;

43/// Rényi divergence of order α: D_α(P || Q) = (1/(α-1)) log Σ p(x)^α q(x)^(1-α)
44///
45/// Generalizes KL divergence. As α → 1, converges to KL.
46///
47/// # Special Cases
48///
49/// - α = 0: -log Q(support(P))
50/// - α = 1/2: Related to Bhattacharyya distance
51/// - α = 1: KL divergence (limit)
52/// - α = 2: Related to χ² divergence
53/// - α → ∞: log(max_x p(x)/q(x))
54///
55/// # Arguments
56///
57/// * `p` - Distribution P
58/// * `q` - Distribution Q
59/// * `alpha` - Order parameter (must not be 1)
60///
61/// # Example
62///
63/// ```rust
64/// use surp::fdiv::renyi_divergence;
65///
66/// let p = [0.5, 0.5];
67/// let q = [0.9, 0.1];
68///
69/// let d_half = renyi_divergence(&p, &q, 0.5);
70/// let d_2 = renyi_divergence(&p, &q, 2.0);
71///
72/// // Higher α emphasizes tail behavior differences
73/// assert!(d_2 > d_half);
74/// ```
75pub fn renyi_divergence(p: &[f64], q: &[f64], alpha: f64) -> f64 {
76    assert_eq!(p.len(), q.len(), "distributions must have same length");
77    assert!(
78        (alpha - 1.0).abs() > 0.01,
79        "α cannot equal 1 (use KL instead)"
80    );
81
82    let sum: f64 = p
83        .iter()
84        .zip(q.iter())
85        .filter(|(&pi, &qi)| pi > EPSILON && qi > EPSILON)
86        .map(|(&pi, &qi)| pi.powf(alpha) * qi.powf(1.0 - alpha))
87        .sum();
88
89    if sum <= 0.0 {
90        return f64::INFINITY;
91    }
92
93    sum.ln() / ((alpha - 1.0) * LN_2)
94}

96/// Chi-squared divergence: χ²(P || Q) = Σ (p(x) - q(x))² / q(x)
97///
98/// Related to Pearson's chi-squared test statistic.
99/// Sensitive to differences where Q is small.
100///
101/// # Properties
102///
103/// - χ²(P || Q) ≥ 0, = 0 iff P = Q
104/// - χ²(P || Q) ≥ 2 * TV(P, Q)² (Pinsker-like)
105/// - Asymmetric: χ²(P || Q) ≠ χ²(Q || P)
106///
107/// # Example
108///
109/// ```rust
110/// use surp::fdiv::chi_squared_divergence;
111///
112/// let p = [0.5, 0.5];
113/// let q = [0.9, 0.1];
114///
115/// let chi2 = chi_squared_divergence(&p, &q);
116/// assert!(chi2 > 0.0);
117/// ```
118pub fn chi_squared_divergence(p: &[f64], q: &[f64]) -> f64 {
119    assert_eq!(p.len(), q.len(), "distributions must have same length");
120
121    p.iter()
122        .zip(q.iter())
123        .filter(|(_, &qi)| qi > EPSILON)
124        .map(|(&pi, &qi)| (pi - qi).powi(2) / qi)
125        .sum()
126}
127
128/// Symmetric chi-squared divergence: (χ²(P||Q) + χ²(Q||P)) / 2
129pub fn chi_squared_symmetric(p: &[f64], q: &[f64]) -> f64 {
130    (chi_squared_divergence(p, q) + chi_squared_divergence(q, p)) / 2.0
131}

133/// Hellinger distance: H(P, Q) = (1/√2) √Σ (√p(x) - √q(x))²
134///
135/// A symmetric, bounded metric on probability distributions.
136///
137/// # Properties
138///
139/// - 0 ≤ H(P, Q) ≤ 1
140/// - H(P, Q) = H(Q, P)
141/// - H² is an f-divergence with f(t) = (√t - 1)²
142///
143/// # Relationship to Other Divergences
144///
145/// - H²(P,Q) ≤ KL(P||Q)
146/// - H²(P,Q) ≤ TV(P,Q) ≤ H(P,Q)√(2 - H²(P,Q))
147/// - H²(P,Q) = 1 - BC(P,Q), where BC is Bhattacharyya coefficient
148///
149/// # Example
150///
151/// ```rust
152/// use surp::fdiv::hellinger_distance;
153///
154/// let p = [0.5, 0.5];
155/// let q = [0.9, 0.1];
156///
157/// let h = hellinger_distance(&p, &q);
158/// assert!(h >= 0.0 && h <= 1.0);
159/// ```
160pub fn hellinger_distance(p: &[f64], q: &[f64]) -> f64 {
161    assert_eq!(p.len(), q.len(), "distributions must have same length");
162
163    let sum: f64 = p
164        .iter()
165        .zip(q.iter())
166        .map(|(&pi, &qi)| (pi.sqrt() - qi.sqrt()).powi(2))
167        .sum();
168
169    (sum / 2.0).sqrt()
170}

172/// Hellinger distance squared (the f-divergence form).
173pub fn hellinger_squared(p: &[f64], q: &[f64]) -> f64 {
174    let h = hellinger_distance(p, q);
175    h * h
176}

178/// Tensorization of squared Hellinger distance for i.i.d. product measures.
179///
180/// If \(H^2(P,Q)\) is the squared Hellinger distance between single-sample distributions,
181/// then for \(n\) i.i.d. samples:
182///
183/// ```text
184/// H^2(P^{⊗n}, Q^{⊗n}) = 1 - (1 - H^2(P,Q))^n
185/// ```
186///
187/// This identity is the workhorse behind two-point testing lower bounds: if
188/// \(H^2(P,Q)\) is \(O(1/n)\), then \(P^{⊗n}\) and \(Q^{⊗n}\) remain hard to distinguish.
189///
190/// # Arguments
191///
192/// * `h2` - Squared Hellinger distance in [0, 1]
193/// * `n` - Number of i.i.d. samples
194pub fn hellinger_squared_tensorized(h2: f64, n: usize) -> f64 {
195    if n == 0 {
196        return 0.0;
197    }
198    // Be conservative: clamp to a valid range rather than panic in a metric helper.
199    let h2 = h2.clamp(0.0, 1.0);
200    if h2 == 0.0 {
201        return 0.0;
202    }
203    if h2 == 1.0 {
204        return 1.0;
205    }
206
207    // Compute 1 - (1 - h2)^n in a numerically stable-ish way.
208    // For small h2 and large n, exp(n * ln(1-h2)) behaves better than powi on f64.
209    let log_term = (1.0 - h2).ln();
210    let a = (n as f64) * log_term; // <= 0
211                                   // 1 - exp(a) = -expm1(a), which is more stable when a is close to 0.
212    (-a.exp_m1()).clamp(0.0, 1.0)
213}

215/// Bhattacharyya coefficient: BC(P, Q) = Σ √(p(x) q(x))
216///
217/// Measures overlap between distributions. BC ∈ [0, 1].
218///
219/// # Properties
220///
221/// - BC(P, Q) = 1 iff P = Q
222/// - BC(P, Q) = 0 iff P and Q have disjoint support
223/// - Related to Hellinger: H²(P,Q) = 1 - BC(P,Q)
224///
225/// # Example
226///
227/// ```rust
228/// use surp::fdiv::bhattacharyya_coefficient;
229///
230/// let p = [0.5, 0.5];
231/// let q = [0.5, 0.5];
232///
233/// let bc = bhattacharyya_coefficient(&p, &q);
234/// assert!((bc - 1.0).abs() < 1e-10);  // Identical distributions
235/// ```
236pub fn bhattacharyya_coefficient(p: &[f64], q: &[f64]) -> f64 {
237    assert_eq!(p.len(), q.len(), "distributions must have same length");
238
239    p.iter()
240        .zip(q.iter())
241        .map(|(&pi, &qi)| (pi * qi).sqrt())
242        .sum()
243}
244
245/// Bhattacharyya distance: -ln(BC(P, Q))
246pub fn bhattacharyya_distance(p: &[f64], q: &[f64]) -> f64 {
247    -bhattacharyya_coefficient(p, q).ln()
248}

250/// Total variation distance: TV(P, Q) = (1/2) Σ |p(x) - q(x)|
251///
252/// The most natural metric on probability distributions.
253///
254/// # Properties
255///
256/// - TV(P, Q) ∈ [0, 1]
257/// - TV(P, Q) = sup_A |P(A) - Q(A)| (variational form)
258/// - TV is a proper metric (satisfies triangle inequality)
259///
260/// # Relationship to Other Divergences
261///
262/// - TV²(P,Q) ≤ KL(P||Q)/2 (Pinsker's inequality)
263/// - TV(P,Q) ≤ H(P,Q) √(2 - H²(P,Q))
264///
265/// # Example
266///
267/// ```rust
268/// use surp::fdiv::total_variation;
269///
270/// let p = [0.5, 0.5];
271/// let q = [0.9, 0.1];
272///
273/// let tv = total_variation(&p, &q);
274/// assert!((tv - 0.4).abs() < 1e-10);  // |0.5-0.9|/2 + |0.5-0.1|/2 = 0.4
275/// ```
276pub fn total_variation(p: &[f64], q: &[f64]) -> f64 {
277    assert_eq!(p.len(), q.len(), "distributions must have same length");
278
279    p.iter()
280        .zip(q.iter())
281        .map(|(&pi, &qi)| (pi - qi).abs())
282        .sum::<f64>()
283        / 2.0
284}

286/// α-divergence: D_α(P || Q) = (1/(α(α-1))) (Σ p^α q^(1-α) - 1)
287///
288/// A family that interpolates between forward and reverse KL.
289///
290/// # Special Cases
291///
292/// - α → 0: KL(Q || P) (reverse KL)
293/// - α = 0.5: Symmetric, related to Hellinger
294/// - α → 1: KL(P || Q) (forward KL)
295///
296/// # Example
297///
298/// ```rust
299/// use surp::fdiv::alpha_divergence;
300///
301/// let p = [0.5, 0.5];
302/// let q = [0.9, 0.1];
303///
304/// let d_half = alpha_divergence(&p, &q, 0.5);
305/// let d_2 = alpha_divergence(&p, &q, 2.0);
306/// ```
307pub fn alpha_divergence(p: &[f64], q: &[f64], alpha: f64) -> f64 {
308    assert_eq!(p.len(), q.len(), "distributions must have same length");
309
310    if alpha.abs() < EPSILON {
311        // α → 0: reverse KL
312        return q
313            .iter()
314            .zip(p.iter())
315            .filter(|(&qi, &pi)| qi > EPSILON && pi > EPSILON)
316            .map(|(&qi, &pi)| qi * (qi / pi).ln())
317            .sum::<f64>()
318            / LN_2;
319    }
320
321    if (alpha - 1.0).abs() < EPSILON {
322        // α → 1: forward KL
323        return p
324            .iter()
325            .zip(q.iter())
326            .filter(|(&pi, &qi)| pi > EPSILON && qi > EPSILON)
327            .map(|(&pi, &qi)| pi * (pi / qi).ln())
328            .sum::<f64>()
329            / LN_2;
330    }
331
332    let sum: f64 = p
333        .iter()
334        .zip(q.iter())
335        .filter(|(&pi, &qi)| pi > EPSILON && qi > EPSILON)
336        .map(|(&pi, &qi)| pi.powf(alpha) * qi.powf(1.0 - alpha))
337        .sum();
338
339    (sum - 1.0) / (alpha * (alpha - 1.0))
340}

342/// Fisher information distance (approximation via Bhattacharyya).
343///
344/// The Fisher information metric defines a Riemannian geometry on
345/// the space of probability distributions. The geodesic distance
346/// under this metric is related to the Hellinger distance:
347///
348/// ```text
349/// d_Fisher ≈ 2 * arccos(BC(P, Q))  (for distributions with same support)
350/// ```
351///
352/// # Connection to Statistical Estimation
353///
354/// Fisher information governs the Cramér-Rao bound for unbiased estimators.
355/// For location families, the Fisher information rate determines minimax
356/// estimation rates. In two-point testing, distinguishing P from Q requires
357/// n = Ω(1/d²_Fisher) samples.
358///
359/// See also: `hellinger_squared_tensorized` for the tensorization identity
360/// that underlies many statistical lower bounds.
361pub fn fisher_distance_approx(p: &[f64], q: &[f64]) -> f64 {
362    let bc = bhattacharyya_coefficient(p, q);
363    2.0 * bc.clamp(-1.0, 1.0).acos()
364}

366/// f-divergence with custom generator function.
367///
368/// D_f(P || Q) = Σ q(x) f(p(x)/q(x))
369///
370/// # Arguments
371///
372/// * `p` - Distribution P
373/// * `q` - Distribution Q
374/// * `f` - Generator function (convex, f(1) = 0)
375///
376/// # Example
377///
378/// ```rust
379/// use surp::fdiv::f_divergence;
380///
381/// let p = [0.5, 0.5];
382/// let q = [0.9, 0.1];
383///
384/// // Chi-squared: f(t) = (t-1)²
385/// let chi2 = f_divergence(&p, &q, |t| (t - 1.0).powi(2));
386/// ```
387pub fn f_divergence<F>(p: &[f64], q: &[f64], f: F) -> f64
388where
389    F: Fn(f64) -> f64,
390{
391    assert_eq!(p.len(), q.len(), "distributions must have same length");
392
393    p.iter()
394        .zip(q.iter())
395        .filter(|(_, &qi)| qi > EPSILON)
396        .map(|(&pi, &qi)| qi * f(pi / qi))
397        .sum()
398}
